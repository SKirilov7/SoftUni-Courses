rows,cols = [int(num) for num in input().split()]

matrix = []

for row in range(rows):
    matrix.append(input().split())

counter_equal_chars = 0

for row in range(0,rows - 1):
    for col in range(0,cols - 1):
        primary_char = matrix[row][col]
        next_to_primary = matrix[row][col + 1]
        above_primary = matrix[row+1][col]
        next_to_above_primary = matrix[row+1][col+1]
        list_chars = [primary_char,next_to_primary,above_primary,next_to_above_primary]

        if len(set(list_chars)) == 1:
            counter_equal_chars += 1

print(counter_equal_chars)
