clothes_amount = [int(num) for num in input().split()]
max_clothes_in_a_rack = int(input())
racks = 0
current_rack_clothes = max_clothes_in_a_rack
while clothes_amount:
    current_amount_of_clothes = clothes_amount[-1]

    if current_rack_clothes >= current_amount_of_clothes:
        current_rack_clothes -= current_amount_of_clothes
        clothes_amount.pop()
        if len(clothes_amount) == 1:
            racks += 1
    else:
        racks += 1
        current_rack_clothes = max_clothes_in_a_rack

print(racks)
