def odd_or_even(com,list_of_nums):
    if com == 'Odd':
        numbers = [num for num in list_of_nums if not num % 2 == 0]
    else:
        numbers = [num for num in list_of_nums if  num % 2 == 0]
    total_sum = sum(numbers) * len(list_of_nums)
    return total_sum

command = input()
numbers = list(map(int,input().split()))

print(odd_or_even(command,numbers))
